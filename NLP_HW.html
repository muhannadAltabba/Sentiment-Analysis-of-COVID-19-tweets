STOPWORDS = set(stopwords.words('english'))
tokenizer = TreebankWordTokenizer()

def text_prepare(text):
    """
        text: a string
        
        return: modified initial string
    """
    wnl = nltk.WordNetLemmatizer()
    text = text.lower()
    text = re.sub(TAG_RE,"",text)
    text = re.sub(LINK_RE,"",text)
    text = re.sub(REPLACE_BY_SPACE_RE,"",text)
    text = re.sub(SPACE_RE," ",text)
    text = re.sub(START_SPACE_RE,"",text)
    text = re.sub(END_SPACE_RE,"",text)
    text = re.sub(DATE_RE,"",text)
    text = re.sub(A_HAT_RE,"a",text)
    text = re.sub(NUMBER_RE,"",text)
    text = re.sub(SPACE_AFTER_HASH_RE,"",text)
    wnl = nltk.WordNetLemmatizer()
    tokens = tokenizer.tokenize(text)
    lemmatizedTokens =[wnl.lemmatize(t) for t in tokens]
    text = " ".join([w for w in lemmatizedTokens if not w in STOPWORDS])# delete stopwords from text
    text = re.sub(SPACE_AFTER_HASH_RE,"#",text)
    return text

detokenize = TreebankWordDetokenizer().detokenize

def generate_sent(model, num_words, random_seed=42):
    """
    :param model: An ngram language model from `nltk.lm.model`.
    :param num_words: Max no. of words to generate.
    :param random_seed: Seed value for random.
    """
    content = []
    for token in model.generate(num_words, random_seed=random_seed):
        if token == '<s>':
            continue
        if token == '</s>':
            break
        content.append(token)
    return detokenize(content)

def init_data(count_of_class):
    lines = []
    sentiment = []
    for twit, sent in zip(tokenized_text,df['Sentiment'].values):
        if(count_of_class == 2):
            if len(twit) > 1 and sent != "Neutral": 
                lines.append(twit)
                if sent == "Extremely Negative" or sent == "Negative":
                    sentiment.append(0)
                elif sent == "Extremely Positive" or sent == "Positive":
                    sentiment.append(1)
        else:
            if len(twit) > 1: 
                lines.append(twit)
                if sent == "Neutral": 
                    sentiment.append(0)
                elif sent == "Negative":
                    sentiment.append(1)
                elif sent == "Extremely Negative":
                    sentiment.append(2)
                elif sent == "Positive":
                    sentiment.append(3)
                elif sent == "Extremely Positive":
                    sentiment.append(4)
    sentiment = np.asarray(sentiment).astype(np.float)
    return lines, sentiment

def bow(sentences_train, sentences_test, y_train, y_test): 
    vectorizer = CountVectorizer()
    vectorizer.fit(sentences_train)

    X_train_bow = vectorizer.transform(sentences_train)
    X_test_bow  = vectorizer.transform(sentences_test)
    classifier_bow = LogisticRegression(max_iter=10000)
    classifier_bow.fit(X_train_bow, y_train)

    score = classifier_bow.score(X_train_bow, y_train)
    print("Train Accuracy:", score)
    score = classifier_bow.score(X_test_bow, y_test)
    print("Test Accuracy:", score)

def tfidf(X_train, X_test,  y_train, y_test):
    """
        X_train, X_val, X_test — samples        
        return TF-IDF vectorized representation of each sample and vocabulary
    """
    # Create TF-IDF vectorizer with a proper parameters choice
    # Fit the vectorizer on the train set
    # Transform the train, test, and val sets and return the result  
    tfidf_vectorizer = TfidfVectorizer(min_df=5, max_df=0.9, ngram_range=(1, 2))
    X_train = tfidf_vectorizer.fit_transform(X_train)
    X_test = tfidf_vectorizer.transform(X_test)
    
    classifier_tfidf = LogisticRegression(max_iter=10000)
    classifier_tfidf.fit(X_train, y_train)

    score = classifier_tfidf.score(X_train, y_train)
    print("Train Accuracy:", score)
    score = classifier_tfidf.score(X_test, y_test)
    print("Test Accuracy:", score)

def plot_history(history):
    acc = history.history['accuracy']
    val_acc = history.history['val_accuracy']
    loss = history.history['loss']
    val_loss = history.history['val_loss']
    x = range(1, len(acc) + 1)

    plt.figure(figsize=(12, 5))
    plt.subplot(1, 2, 1)
    plt.plot(x, acc, 'b', label='Training acc')
    plt.plot(x, val_acc, 'r', label='Validation acc')
    plt.title('Training and validation accuracy')
    plt.legend()
    plt.subplot(1, 2, 2)
    plt.plot(x, loss, 'b', label='Training loss')
    plt.plot(x, val_loss, 'r', label='Validation loss')

STOPWORDS = set(stopwords.words('english'))

def text_prepare(text):
    wnl = nltk.WordNetLemmatizer()
    text = text.lower()
    text = re.sub(TAG_RE,"",text)
    text = re.sub(LINK_RE,"",text)
    text = re.sub(REPLACE_BY_SPACE_RE,"",text)
    text = re.sub(SPACE_RE," ",text)
    text = re.sub(START_SPACE_RE,"",text)
    text = re.sub(END_SPACE_RE,"",text)
    text = re.sub(DATE_RE,"",text)
    text = re.sub(A_HAT_RE,"a",text)
    text = re.sub(NUMBER_RE,"",text)
    text = re.sub(SPACE_AFTER_HASH_RE,"",text)
    wnl = nltk.WordNetLemmatizer()
    tokens = tokenizer.tokenize(text)
    lemmatizedTokens =[wnl.lemmatize(t) for t in tokens]
    text = " ".join([w for w in lemmatizedTokens if not w in STOPWORDS])# delete stopwords from text
    text = re.sub(SPACE_AFTER_HASH_RE,"#",text)
    return text

def init_data(tokenized_text, df):
    lines = []
    sentiment = []
    for twit, sent in zip(tokenized_text,df['Sentiment'].values):
        if len(twit) > 1: 
            lines.append(twit)
            if sent == "Neutral": 
                sentiment.append(0)
            elif sent == "Negative":
                sentiment.append(1)
            elif sent == "Extremely Negative":
                sentiment.append(2)
            elif sent == "Positive":
                sentiment.append(3)
            elif sent == "Extremely Positive":
                sentiment.append(4)
    sentiment = np.asarray(sentiment).astype(np.float)
    return lines, sentiment
